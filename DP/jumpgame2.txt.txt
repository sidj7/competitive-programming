class Solution {
public:
    int help(int t,int n,vector<int>&dp,vector<int>&nums)
    {
        if(t==n-1)return 0;
        else if(t>=n)return 1e9;
        else if(dp[t]!=-1)return dp[t];
        int mn=1e9;
        for(int i=1;i<=nums[t];i++)
        {
            int j=1+help(t+i,n,dp,nums);
            mn=min(mn,j);
        }
        return dp[t]=mn;
    }
    int jump(vector<int>& nums) {
        int n=nums.size();
        vector<int>dp(n,-1);
        return help(0,n,dp,nums);
    }
};
