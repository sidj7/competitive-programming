class Solution {
public:
    string convert(string s, int n) {
        if(n<=1)return s;
        vector<string>res(n,"");
        int j=0,d=-1;
        for(auto i:s)
        {
            res[j]+=i;
            if(d==1)j--;
            else j++;
            if(j==0 || j==n-1) d*=(-1);
        }
        string ans="";
        for(auto x:res)ans+=x;
        return ans;
    }
};
